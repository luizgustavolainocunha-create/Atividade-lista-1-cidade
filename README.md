# Atividade-lista-1-cidade
programa {

    cadeia nomeAluno
    inteiro idade
    cadeia cidade
    cadeia escola

    funcao inicio() {

        escreva("Informe o seu nome: ")
        leia(nomeAluno)

        escreva("\nInforme sua idade: ")
        leia(idade)

        escreva("\nOnde você mora? ")
        leia(cidade)

        escreva("\nOnde você estuda? ")
        leia(escola)

        escreva("\n----- DADOS INFORMADOS -----\n")
        escreva("Nome: ", nomeAluno, "\n")
        escreva("Idade: ", idade, "\n")
        escreva("Mora em: ", cidade, "\n")
        escreva("Estuda em: ", escola, "\n")
    }

}
